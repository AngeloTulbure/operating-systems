import java.util.Random;



public class Cliente extends Thread {
	String tipo; //CER o IDE
	int id;
	Monitor m;
	Random r;
	
	public Cliente(String tipo, int id, Monitor m, Random r ) {
		this.tipo = tipo;
		this.id = id;
		this.m = m;
		this.r=r;
	}

	public String getTipo() {
		return tipo;
	}

	public int getIdCliente() {
		return id;
	}

	public void run() {
		try {
			Thread.sleep(r.nextInt(150)); 
			m.richiediServizio(this);
			Thread.sleep(r.nextInt(200)); //tempo per l'erogazione del servizio
			m.fineServizio(this); 
			System.out.println(this.id+": "+this.tipo +" terminato");
		} catch (InterruptedException e) {
			e.printStackTrace();
		}
	}	
}
